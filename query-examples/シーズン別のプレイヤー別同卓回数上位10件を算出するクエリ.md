## 説明
シーズン別のプレイヤー別同卓回数上位10件を算出するクエリ

## SQL クエリ
```sql
-- シーズンごとのプレイヤーペア別同卓回数と勝敗を集計する
WITH season_cotable AS (
    SELECT
        ls.start_year,
        gpr1.player_id AS player1_id,
        gpr2.player_id AS player2_id,
        COUNT(*) AS cotable_count,
        -- 勝敗集計（順位が小さい方が勝ち）
        SUM(CASE WHEN gpr1.rank < gpr2.rank THEN 1 ELSE 0 END) AS player1_win_count,
        SUM(CASE WHEN gpr1.rank > gpr2.rank THEN 1 ELSE 0 END) AS player2_win_count
    FROM game_player_result gpr1
    JOIN game_player_result gpr2
        ON gpr1.game_id = gpr2.game_id
        AND gpr1.player_id < gpr2.player_id
    -- 試合・シーズン情報の結合
    JOIN game g ON gpr1.game_id = g.id
    JOIN season_stage ss ON g.season_stage_id = ss.id
    JOIN league_season ls ON ss.league_season_id = ls.id
    GROUP BY ls.start_year, gpr1.player_id, gpr2.player_id
),
-- シーズン内で同卓回数の多い順に10件を選出する
ranked AS (
    SELECT *,
        ROW_NUMBER() OVER (PARTITION BY start_year ORDER BY cotable_count DESC, player1_id, player2_id) AS row_num
    FROM season_cotable
)
-- 上位10件のペアにプレイヤー名を付与して出力する
SELECT
    start_year || '-' || (start_year + 1) AS シーズン,
    row_num AS 順位,
    p1.name AS プレイヤー1,
    p2.name AS プレイヤー2,
    cotable_count AS 同卓回数,
    player1_win_count AS プレイヤー1勝ち数,
    player2_win_count AS プレイヤー2勝ち数
FROM ranked
-- プレイヤー名の結合
JOIN player p1 ON player1_id = p1.id
JOIN player p2 ON player2_id = p2.id
WHERE row_num <= 10
ORDER BY start_year, row_num;
```
