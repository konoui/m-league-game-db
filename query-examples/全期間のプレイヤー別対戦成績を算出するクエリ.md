## 説明
全期間のプレイヤー別対戦成績を算出するクエリ

## SQL クエリ
```sql
-- 同一試合の全プレイヤーペアの対戦成績を集計する
WITH head_to_head AS (
    SELECT
        gpr1.player_id AS player1_id,
        gpr2.player_id AS player2_id,
        -- 対戦回数と勝敗の集計
        COUNT(*) AS match_count,
        SUM(CASE WHEN gpr1.rank < gpr2.rank THEN 1 ELSE 0 END) AS player1_win_count,
        SUM(CASE WHEN gpr1.rank > gpr2.rank THEN 1 ELSE 0 END) AS player1_lose_count,
        SUM(CASE WHEN gpr1.rank = gpr2.rank THEN 1 ELSE 0 END) AS draw_count,
        -- ポイント合計の集計
        SUM(gpr1.points) AS player1_total_points,
        SUM(gpr2.points) AS player2_total_points
    FROM game_player_result gpr1
    JOIN game_player_result gpr2
        ON gpr1.game_id = gpr2.game_id
        AND gpr1.player_id < gpr2.player_id
    GROUP BY gpr1.player_id, gpr2.player_id
)
-- 最終結果: プレイヤー名と勝ち越し判定を付与して出力
SELECT
    p1.name AS プレイヤー1,
    p2.name AS プレイヤー2,
    h.match_count AS 対戦回数,
    h.player1_win_count AS プレイヤー1勝ち回数,
    h.player1_lose_count AS プレイヤー1負け回数,
    h.draw_count AS 引き分け回数,
    ROUND(h.player1_total_points, 1) AS プレイヤー1合計ポイント,
    ROUND(h.player2_total_points, 1) AS プレイヤー2合計ポイント,
    ROUND(h.player1_total_points - h.player2_total_points, 1) AS ポイント差,
    CASE
        WHEN h.player1_win_count > h.player1_lose_count THEN 'プレイヤー1勝ち越し'
        WHEN h.player1_win_count < h.player1_lose_count THEN 'プレイヤー2勝ち越し'
        ELSE 'イーブン'
    END AS 勝敗判定
FROM head_to_head h
-- プレイヤー名の結合
JOIN player p1 ON h.player1_id = p1.id
JOIN player p2 ON h.player2_id = p2.id
ORDER BY p1.name, p2.name;
```
