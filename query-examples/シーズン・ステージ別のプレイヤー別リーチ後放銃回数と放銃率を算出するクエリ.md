## 説明
シーズン・ステージ別のプレイヤー別リーチ後放銃回数と放銃率を算出するクエリ

## SQL クエリ
```sql
-- 受理されたリーチごとに同一局内で放銃したかを判定する
WITH reach_dealin AS (
    SELECT
        re.actor_player_id,
        e.kyoku_id,
        CASE
            WHEN EXISTS (
                SELECT 1
                FROM event e2
                JOIN agari_event ae ON e2.id = ae.event_id
                WHERE e2.kyoku_id = e.kyoku_id
                  AND ae.target_player_id = re.actor_player_id
            ) THEN 1
            ELSE 0
        END AS is_dealin
    FROM reach_event re
    JOIN event e ON re.event_id = e.id
    WHERE re.is_accepted = 1
),
-- シーズン・ステージ・プレイヤー別にリーチ後放銃統計を集計する
player_reach_dealin_stats AS (
    SELECT
        ls.start_year,
        ss.stage,
        p.name AS player_name,
        t.name AS team_name,
        -- リーチ回数と放銃回数の集計
        COUNT(*) AS reach_count,
        SUM(rd.is_dealin) AS dealin_after_reach_count,
        ROUND(SUM(rd.is_dealin) * 100.0 / COUNT(*), 2) AS dealin_after_reach_rate
    FROM reach_dealin rd
    -- 局・試合・シーズン情報の結合
    JOIN kyoku k ON rd.kyoku_id = k.id
    JOIN game g ON k.game_id = g.id
    JOIN season_stage ss ON g.season_stage_id = ss.id
    JOIN league_season ls ON ss.league_season_id = ls.id
    -- プレイヤー・チーム情報の結合
    JOIN player p ON rd.actor_player_id = p.id
    JOIN player_team pt ON p.id = pt.player_id
        AND ls.start_year >= pt.joined_season_year
        AND ls.start_year <= pt.left_season_year
    JOIN team t ON pt.team_id = t.id
    GROUP BY ls.start_year, ss.stage, p.id, p.name, t.name
)
-- 最終結果: リーチ後放銃回数と放銃率を出力
SELECT
    start_year || '-' || (start_year + 1) AS シーズン,
    stage AS ステージ,
    player_name AS プレイヤー名,
    team_name AS チーム名,
    reach_count AS リーチ回数,
    dealin_after_reach_count AS リーチ後放銃回数,
    dealin_after_reach_rate AS リーチ後放銃率
FROM player_reach_dealin_stats
ORDER BY start_year DESC, stage, dealin_after_reach_rate DESC;
```
