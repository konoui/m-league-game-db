## 説明
全期間のプレイヤー別対戦したことのない選手の組み合わせを取得するクエリ

## SQL クエリ
```sql
-- シーズンごとに在籍していた選手とチームを特定する
WITH season_active_players AS (
    SELECT
        ls.start_year,
        pt.player_id,
        pt.team_id
    FROM league_season ls
    JOIN player_team pt
        ON ls.start_year >= pt.joined_season_year
        AND ls.start_year <= pt.left_season_year
),
-- 同シーズンに在籍し、かつ別チームの一方向ペアを生成する
season_pairs AS (
    SELECT
        sp1.start_year,
        sp1.player_id AS player1_id,
        sp2.player_id AS player2_id
    FROM season_active_players sp1
    JOIN season_active_players sp2
        ON sp1.start_year = sp2.start_year
        AND sp1.player_id < sp2.player_id
        AND sp1.team_id <> sp2.team_id
),
-- シーズンごとに実際に対戦したペアを抽出する
season_faced_pairs AS (
    SELECT DISTINCT
        ls.start_year,
        gpr1.player_id AS player1_id,
        gpr2.player_id AS player2_id
    FROM game_player_result gpr1
    JOIN game_player_result gpr2
        ON gpr1.game_id = gpr2.game_id
        AND gpr1.player_id < gpr2.player_id
    -- 試合・シーズン情報の結合
    JOIN game g ON gpr1.game_id = g.id
    JOIN season_stage ss ON g.season_stage_id = ss.id
    JOIN league_season ls ON ss.league_season_id = ls.id
)
-- 未対戦ペアのみを取得してプレイヤー名を付与する
SELECT
    sp.start_year || '-' || (sp.start_year + 1) AS シーズン,
    p1.name AS プレイヤー1,
    p2.name AS プレイヤー2
FROM season_pairs sp
-- プレイヤー名の結合
JOIN player p1 ON sp.player1_id = p1.id
JOIN player p2 ON sp.player2_id = p2.id
WHERE NOT EXISTS (
    SELECT 1
    FROM season_faced_pairs sfp
    WHERE sfp.start_year = sp.start_year
      AND sfp.player1_id = sp.player1_id
      AND sfp.player2_id = sp.player2_id
)
ORDER BY sp.start_year, p1.name, p2.name;
```
